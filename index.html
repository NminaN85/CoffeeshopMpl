const BASE_URL = "https://docs.google.com/spreadsheets/d/11-u2UCKiW1VmYhm3X8MhD_f4PhyfdXi0DtJG41zdHT8/gviz/tq?tqx=out:json"; // إخفاء شاشة التحميل فوراً وبشكل مضمون بمجرد ما الصفحة تحمل function hideLoader() { const loader = document.getElementById('loading-screen'); if (loader) { loader.classList.add('fade-out'); setTimeout(() => { loader.style.display = 'none'; }, 500); } } window.addEventListener('load', hideLoader); setTimeout(hideLoader, 1000); // دالة إظهار وإخفاء الباسوورد عند الضغط على العين function togglePass() { const passInput = document.getElementById('access-key'); const eyeIcon = document.querySelector('#toggle-password i'); if (passInput.type === "password") { passInput.type = "text"; eyeIcon.classList.remove('fa-eye'); eyeIcon.classList.add('fa-eye-slash'); } else { passInput.type = "password"; eyeIcon.classList.remove('fa-eye-slash'); eyeIcon.classList.add('fa-eye'); } } // دالة التحقق الذكي من الكود من الجوجل شيت async function handleLogin(e) { e.preventDefault(); const codeInput = document.getElementById('access-key').value.trim(); const errDiv = document.getElementById('error-message'); const btn = document.querySelector('.btn-submit span'); errDiv.style.display = 'none'; btn.innerText = "AUTHENTICATING..."; // 1. الكود العام الموحد (تم تعديله لـ MPL2026 بناءً على رغبتك) if(codeInput === "MPL2026") { localStorage.setItem('mocha_auth', 'true'); localStorage.setItem('user_role', 'public'); window.location.href = 'dashboard.html'; return; } try { // 2. فحص الأكواد الفردية للملاك من شيت Investors const response = await fetch(BASE_URL + "&sheet=Investors&nocache=" + new Date().getTime()); const text = await response.text(); const json = JSON.parse(text.substr(47).slice(0, -2)); const rows = json.table.rows; let foundInvestor = null; for (let i = 1; i < rows.length; i++) { const row = rows[i]; if (row && row.c && row.c[2]) { const sheetCode = row.c[2].v ? row.c[2].v.toString().trim() : ""; if (sheetCode === codeInput) { foundInvestor = { name: row.c[0] ? row.c[0].v : "Investor", shares: row.c[1] ? parseFloat(row.c[1].v || 0) : 0 }; break; } } } if (foundInvestor) { localStorage.setItem('mocha_auth', 'true'); localStorage.setItem('user_role', 'investor'); localStorage.setItem('investor_name', foundInvestor.name); localStorage.setItem('investor_shares', foundInvestor.shares); btn.innerText = "ACCESS GRANTED"; window.location.href = 'dashboard.html'; } else { btn.innerText = "VERIFY ACCESS"; errDiv.style.display = 'block'; } } catch (err) { console.error("Auth System Error:", err); btn.innerText = "VERIFY ACCESS"; alert("❌ حدث خطأ أثناء الاتصال بقاعدة البيانات."); } }
